def pandas_onehot(df, cols):
    """ 
    @param df pandas DataFrame
    @param cols a list of columns to encode
    @return a DataFrame with one-hot encoding
    """
    ohe_data = pd.get_dummies(df[cols])
    
    df = df.drop(cols, axis=1)
    df = df.join(ohe_data)
    return df

def proba_to_pred(y_prob, n_classes):
    best_class = 0
    best_proba = 0.0
    y_pred = np.zeros((len(y_prob[0]), n_classes))
    for j in range(len(y_prob[0])):
        best_proba = 0
        for i in range(n_classes):
            if y_prob[i][j][1] > best_proba:
                best_class = i
                best_proba = y_prob[i][j][1]
        y_pred[j][best_class] = 1
    return y_pred

def proba_to_kaggle(y_prob):
    y_pred = np.zeros((len(y_prob[0]), len(y_prob)))
    for j in range(len(y_prob[0])):
        for i in range(len(y_prob)):
            y_pred[j][i] = y_prob[i][j][1] 
    print("Previsao com sucesso")
    return y_pred

from datetime import date, datetime, time

def get_season(date):
    year = date.year
    seasons = {'Summer':(datetime(year,6,21), datetime(year,9,22)),
           'Autumn':(datetime(year,9,23), datetime(year,12,20)),
           'Spring':(datetime(year,3,21), datetime(year,6,20))}
    
    for season,(season_start, season_end) in seasons.items():
        if date>=season_start and date<= season_end:
            return season
    else:
        return 'Winter'

def get_day_period(time_):
    periods = {'Morning':(time(6), time(12)),
           'Evening':(time(12), time(18)),
           'Night':(time(18), time(23,59)), 
           'Small-hours':(time(0), time(6))}
    
    for period,(period_start, period_end) in periods.items():
        if time_>=period_start and time_<= period_end:
            return period

def add_time_feats(df):
    #Add Year
    df["year"] = df["Dates"].apply(lambda x : int(str(x)[:4]))
    #Add Month
    df["month"] = df["Dates"].apply(lambda x : int(str(x)[5:7]))
    #Add day
    df["day"] = df["Dates"].apply(lambda x : int(str(x)[8:10]))
    #Add Season
    df["season"] = df.apply(lambda x : get_season(datetime(x["year"], x["month"], x["day"])), axis = 1)
    #Add Period
    df["day_period"] = df.apply(lambda x : get_day_period(time(int(str(x["Dates"])[11:13]), int(str(x["Dates"])[14:16]))), axis = 1)
    #df["day_period"] = df.apply(lambda x : get_day_period(time(int(str(x["Dates"])[11, 13]) , 20)) , axis = 1)
    ##Add Hour
    df["hour"] = df.apply(lambda x : int(str(x["Dates"])[11:13]), axis = 1)
    ##Add Hour
    df["minute"] = df.apply(lambda x : int(str(x["Dates"])[14:16]), axis = 1)
    #Working day
    df["workday"] = df.apply(lambda x : 0 if x["DayOfWeek"] == "Sunday" or x["DayOfWeek"] == "Saturday" else 1, axis = 1)
    #30-minute
    df["15min"] = df.apply(lambda x : 1 if abs(int(str(x["Dates"])[14:16]) % 15) == 0 else 0, axis = 1)
    
def add_local_feats(df):
    df["avenue"] = df["Address"].apply(lambda x : 1 if str(x)[-2:]=="AV" else 0)
