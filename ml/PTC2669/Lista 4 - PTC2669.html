class Neuron:
    
    def __init__(self, function, intercept):
        self.weights = None
        self.function = function
        self.intercept = intercept
        self.inputs = None
        self.neuron_output = None
        self.fi_error = None
        
    def step_function(self, value):
        if value >= 0: return 1
        else: return -1
    
    def tanh(self, value):
        return (np.exp(value) - np.exp(-value))/(exp(value) + exp(-value))
    
    def activation_function(self, value, function):
        if function == 1:
            return self.step_function(value)
        if function == 2:
            return self.tanh(value)
     
    def activate_func_derivative(self, i):
        if self.function == 2:  
            return (1 - np.power(self.activation_function(self.net(self.inputs[i], self.weights), self.function), 2))
    
    def net(self, point_features, weights):
        return np.dot(np.transpose(point_features), weights)
    
    def output(self, i):
        if(self.weights == None):
            self.weights = np.zeros(len(self.inputs[0]))
            self.weights.fill(0.5)
        return self.activation_function(self.net(self.inputs[i], self.weights), self.function)
    
    def calcAllOutputs(self):
        self.neuron_output = []
        for i in range(len(self.inputs)):
            self.neuron_output.append(self.output(i))
        
    def getWeights(self):
        return self.weights
    
    def setWeights(self, delta_weights):
        for i in range(len(self.weights)):
            self.weights[i] += delta_weights[i] 
    
    def getInputs(self, i):
        return self.inputs[i]
    
    def setInputs(self, inputs):
        self.inputs = inputs
        if self.intercept:
            self.inputs = [[1] + x for x in self.inputs]
       
    def setFiError(self, new):
        self.fi_error = new
    
    def getFiError(self, i):
        return self.fi_error[i]
    
    def getNeuronOutput(self, i):
        return self.neuron_output[i]

class MultiLayerPerceptron():

    def feedFoward(self, feature_matrix):
        hidden_inputs = []
        output_inputs = []
        #defining inputs for each hidden layer
        for hidden in range(self.ml[1]):
            hidden_inputs = []
            for i in range(len(feature_matrix)):
                inputs_in_i = []
                for j in range(self.ml[0]):
                    if(self.connection_matrix[self.ml[0] + hidden][j] == 1):
                        inputs_in_i.append(self.units[j].output(i))
                hidden_inputs.append(inputs_in_i)  
                self.units[self.ml[0]+hidden].setInputs(hidden_inputs)
                self.units[self.ml[0]+hidden].output(i)        
           
            self.units[self.ml[0]+hidden].setInputs(hidden_inputs)
            self.units[self.ml[0]+hidden].calcAllOutputs()
           
        for output_unit in range(self.ml[2]):
            output_inputs = []
            for i in range(len(feature_matrix)):
                inputs_in_i = []
                
                #looking for inputs from input units
                for j in range(self.ml[0]):
                    if(self.connection_matrix[self.ml[0] + self.ml[1] + output_unit][j] == 1):
                        inputs_in_i.append(self.units[j].output(i))
                for j in range(self.ml[1]):
                    if(self.connection_matrix[self.ml[0] + self.ml[1] + output_unit][j+self.ml[0]] == 1):
                        inputs_in_i.append(self.units[j+self.ml[0]].output(i))   
                output_inputs.append(inputs_in_i) 
                self.units[self.ml[0] + self.ml[1] + output_unit].setInputs(output_inputs)
                self.units[self.ml[0] + self.ml[1] + output_unit].output(i)

            self.units[self.ml[0] + self.ml[1] + output_unit].setInputs(output_inputs)
            self.units[self.ml[0] + self.ml[1] + output_unit].output(i)
            self.units[self.ml[0] + self.ml[1] + output_unit].calcAllOutputs()

    def error(self, i, unit, expected):
        return expected - unit.output(i)

    def predict(self, inputs):
        for i in range(self.ml[0]):
            self.units[i].defInput(inputs[:, i])
        self.feedFoward(inputs)

    def backPropagation(self, feature_matrix, labels, step_size):
        alpha= 0.05
        
        for output_unit in range(self.ml[2]):
            unit = self.units[self.ml[0] + self.ml[1] + output_unit]
            fi = []
            last_delta = [0]
            for i in range(len(feature_matrix)):
                unit_error = self.error(i, unit, labels[i])
                fi.append(unit_error*unit.activate_func_derivative(i))
                delta = [x*step_size for x in unit.inputs[i]]
                delta = [x*fi[i] for x in delta]
                last_delta = [x*alpha for x in last_delta]
                delta = np.array(last_delta) + np.array(delta)
                unit.setWeights(delta)
                last_delta = delta
            unit.setFiError(fi)
        
        for hidden in range(self.ml[1]):
